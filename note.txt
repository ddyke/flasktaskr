db flow

views.py

app = Flask(__name__)                   # create a Flask app
app.config.from_object('_config')       # configure the app
db = SQLAlchemy(app)                    # create SQLAlchemy database bound to Flask app

then models.py

class Task(db.Model):                   # define a 'tasks' table and its values and define print format
class User(db.Model):                   # define a 'users' table and its values and define print format

then views.py

from models import Task, User           # link back the db classes
                                        # the sql tables are not created at this point
                                        # to create the tables call db.create_all()
